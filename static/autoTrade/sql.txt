select symbol,group_concat(value ORDER BY value separator ',,') as myList  from (
select symbol,min(CONVERT(step_value,double)) as 'value' from nse_stock_strike_intervals where month_type='M1' group by symbol 
union
select symbol,max(CONVERT(step_value,double)) as 'value' from nse_stock_strike_intervals where month_type='M1' group by symbol
) 
a group by symbol order by symbol;
